23  @Component
24  @Log4j2
25  /**
26   * Service to add logique after a succesfull authentication with OAuth2:
27   * if user exist then we update it with client provider and its id provider.
28   * Else with create a new user and save it with infirmations given from
29   * Oauth2authentication
30   * Use of SavedRequestAwareAuthenticationSuccessHandler and not
31   * SimpleUrlAuthenticationSuccessHandler with this implementation springboot
32   * save the first url just before login and redirect to it
33   */
34  public class OAuth2LoginSuccessHandler extends SavedRequestAwareAuthenticationSuccessHandler {
35  
36      @Autowired
37      private UserService userService;
38  
39      @Autowired
40      private MessageSource messageSource;
41  
42      @Override
43      public void onAuthenticationSuccess(HttpServletRequest request, HttpServletResponse response,
44              Authentication authentication) throws IOException, ServletException {
45  
46          CustomOAuth2User oAuth2User = (CustomOAuth2User) authentication.getPrincipal();
47  
48          User existingUser = userService.findByUsername(oAuth2User.getEmail());
49  
50          if (existingUser != null) {
51  
52              // update of user with providerId and authenticationProvider if not already done
53              log.info(messageSource.getMessage("global.existing-user.oauth2-authenticated",
54                      new Object[] { existingUser }, LocaleContextHolder.getLocale()));
55  
56              if (existingUser.getAuthenticationProvider() == AuthProvider.LOCAL) {
57  
58                  userService.updateUserFromOAuth2Authentication(oAuth2User, existingUser);
59  
60              } else if ((!Objects.equals(existingUser.getIdProvider(), oAuth2User.getproviderId())
61                      || existingUser.getAuthenticationProvider() != oAuth2User.getClientProvider())) {
62  
63                  throw new OAuth2AuthenticationException("a problem occured with Oauth2Authentication!");
64              }
65  
66          } else {
67  
68              // creation of new user
69              log.info(messageSource.getMessage("global.not-existing-user.oauth2-authenticated",
70                      new Object[] { "createdUser" }, LocaleContextHolder.getLocale()));
71              userService.saveUserFromOAuth2Authentication(oAuth2User);
72          }
73  
74          super.onAuthenticationSuccess(request, response, authentication);
75      }

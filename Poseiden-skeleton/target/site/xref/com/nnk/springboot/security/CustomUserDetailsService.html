31      @Override
32      public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
33  
34          Optional<com.nnk.springboot.domain.User> user = userRepository.findByUsername(username);
35  
36          if (user.isPresent()) {
37              List<SimpleGrantedAuthority> userRoles = new ArrayList<>();
38              userRoles.add(new SimpleGrantedAuthority(user.get().getRole()));
39              return new User(user.get().getUsername(), user.get().getPassword(), userRoles);
40          } else {
41              throw new UsernameNotFoundException(messageSource.getMessage("global.user.not-found", new Object[] { username }, LocaleContextHolder.getLocale()));
42          }
43      }

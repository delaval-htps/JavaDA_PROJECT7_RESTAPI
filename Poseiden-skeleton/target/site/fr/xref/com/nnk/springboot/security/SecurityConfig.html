16  @EnableWebSecurity

18  public class SecurityConfig {
19      @Autowired
20      private CustomUserDetailsService userDetailsService;
21  
22      @Autowired
23      private CustomOauth2UserService oauth2UserService;
24  
25      @Autowired
26      private OAuth2LoginSuccessHandler oAuth2LoginSuccessHandler;

28      @Bean
29      public DaoAuthenticationProvider authenticationProvider() {
30          DaoAuthenticationProvider authProvider = new DaoAuthenticationProvider();
31          authProvider.setUserDetailsService(userDetailsService);
32          authProvider.setPasswordEncoder(passwordEncoder());
33          return authProvider;
34      }
35  
36      @Bean
37      public PasswordEncoder passwordEncoder() {
38          return new BCryptPasswordEncoder();
39      }
40  
41      @Bean
42      public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
43  
44          http.authorizeRequests()
45  
46                  .antMatchers("/css/**")
47                  .permitAll()
48  
49                  .antMatchers("/")
50                  .permitAll()
51  
52                  /*
53                   * don't use hasRole("ADMIN") cause Role in user is "ADMIN" and
54                   * not "ROLE_ADMIN" , use hasAuthority("ADMIN").In spring
55                   * security, hasRole() is the same as hasAuthority(), but
56                   * hasRole() function map with Authority without ROLE_ prefix.
57                   */
58                  .antMatchers("/bidList/**", "/curvePoint/**", "/rating/**", "ruleName/**", "/trade/**", "/user/home")
59                  .hasAnyAuthority("USER", "ADMIN")
60  
61                  .antMatchers("/user/**", "/admin/**")
62                  .hasAuthority("ADMIN")
63  
64                  .anyRequest().authenticated()
65                  
66                  .and()
67                  // using AccesDeniedPage make easy to redirect to error 403 page
68                  .exceptionHandling().accessDeniedPage("/app/error");
69  
70          http.formLogin()
71                  .loginPage("/app/login")
72                  .loginProcessingUrl("/process-login")
73                  .permitAll();
74  
75          http.oauth2Login()
76                  .loginPage("/app/login")
77                  .userInfoEndpoint()
78                  .userService(oauth2UserService)
79                  .and()
80                  .successHandler(oAuth2LoginSuccessHandler);
81  
82          http.logout()
83                  .logoutUrl("/app-logout")
84                  .logoutSuccessUrl("/")
85                  .permitAll();
86  
87          return http.build();
88      }
